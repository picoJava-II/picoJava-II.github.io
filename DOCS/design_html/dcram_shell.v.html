`include "defines.h"

module dcram_shell( clk, bist_mode, bist_reset, test_mode,
		    data_in, we, bypass, bank_sel, reg_enable, addr,
		    sin, so, sm, data_out, dcache_test_err_l);

input clk;
input [1:0] bist_mode;
input bist_reset;
input test_mode;
input [63:0] data_in;
input [3:0] we;
input bypass;
input [1:0] bank_sel;
input reg_enable;
input [`dc_msb:0] addr;
input sin;
input sm;
output so;
output [63:0] data_out;
output dcache_test_err_l;

wire [63:0] bist_data_in;
wire [3:0] bist_we;
wire [`dc_msb-2:0] bist_addr;

dcram_top dcram_top (	.data_in(data_in),
			.we(we),
			.bypass(bypass),
			.bank_sel(bank_sel),
			.enable(reg_enable),
			.addr(addr),
			.clk(clk),
			.data_out(data_out),
			.bist_reset(bist_reset),
			.bist_mode(bist_mode),
			.sin(sin),
			.sm(sm),
			.so(so),
			.test_mode(test_mode),
			.ERROR(dcache_test_err_l));

dcram_misc dcram_misc(	.addr(addr),
			.bank_sel(bank_sel),
			.bypass(bypass),
			.sin(sin),
			.sm(sm),
			.so(so),
			.we(we),
			.data_in(data_in),
			.clk(clk));
endmodule
