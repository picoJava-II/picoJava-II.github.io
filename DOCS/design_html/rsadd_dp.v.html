			r0out,
			stin, 
			sticky,

			b1, 
                  	b0, 
			bsmd, 
			aqcin, 
			rsout, 
			rsovfi 
			);

input  [31:0] r0out, r1out;
//input   [1:0] r0md, r1md;

//input         a2;
input   [4:0] saout;
input  [31:0] b1;

input   [1:0] bsmd;

output [31:0] rsout;
