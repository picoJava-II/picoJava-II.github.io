      input current_state;
      input valid;
      input hold;
      reg   next_state;

      parameter IDLE = 1'b0,
		HOLD = 1'b1;

      begin : fsm
	 case (current_state)
	   IDLE:
	      if (hold & valid)
		 next_state = HOLD;
	      else
		 next_state = current_state;
	   HOLD:

		 next_state = IDLE;
	      else
		 next_state = current_state;
	   default:
	      next_state = 1'bx;
	 endcase // case(current_state)
	 load_buffer_fsm = next_state;
