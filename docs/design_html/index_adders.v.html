module index1_add (

operand,
sum
);

input	[3:0]	operand;
output	[7:0]	sum;


assign	sum[7] = 1'b0;
assign	sum[6] = 1'b0;
assign	sum[5] = 1'b0;
assign	sum[1] = 1'b0;

assign	sum[4] =  operand[3];
assign	sum[3] =  operand[2];
assign	sum[2] =  operand[1];

// If the operand is zero i.e 4'b0001, then sum should also be
// zero i.e 7'b0000001

assign	sum[0] =  operand[0];

endmodule
